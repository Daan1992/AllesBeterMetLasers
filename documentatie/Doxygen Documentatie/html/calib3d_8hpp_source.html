310 /* Block matching algorithm structure */
311 typedef struct CvStereoBMState
312 {
313  // pre-filtering (normalization of input images)
314  int preFilterType; // =CV_STEREO_BM_NORMALIZED_RESPONSE now
315  int preFilterSize; // averaging window size: ~5x5..21x21
316  int preFilterCap; // the output of pre-filtering is clipped by [-preFilterCap,preFilterCap]
317 
318  // correspondence using Sum of Absolute Difference (SAD)
319  int SADWindowSize; // ~5x5..21x21
320  int minDisparity; // minimum disparity (can be negative)
321  int numberOfDisparities; // maximum disparity - minimum disparity (> 0)
322 
323  // post-filtering
324  int textureThreshold; // the disparity is only computed for pixels
325  // with textured enough neighborhood
326  int uniquenessRatio; // accept the computed disparity d* only if
327  // SAD(d) >= SAD(d*)*(1 + uniquenessRatio/100.)
328  // for any d != d*+/-1 within the search range.
329  int speckleWindowSize; // disparity variation window
330  int speckleRange; // acceptable range of variation in window
331 
332  int trySmallerWindows; // if 1, the results may be more accurate,
333  // at the expense of slower processing
334  CvRect roi1, roi2;
335  int disp12MaxDiff;
336 
337  // temporary buffers
338  CvMat* preFilteredImg0;
339  CvMat* preFilteredImg1;
340  CvMat* slidingSumBuf;
341  CvMat* cost;
342  CvMat* disp;
343 } CvStereoBMState;
